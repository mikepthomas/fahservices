38      /** {@inheritDoc} */
39      @Override
40      public Set<Class<?>> getClasses() {
41          Set<Class<?>> resources = new java.util.HashSet<>();
42          resources.add(com.wordnik.swagger.jaxrs.listing.ApiListingResource.class);
43          resources.add(com.wordnik.swagger.jaxrs.listing.ApiDeclarationProvider.class);

45          resources.add(com.wordnik.swagger.jaxrs.listing.ResourceListingProvider.class);
46          addRestResourceClasses(resources);
47          return resources;
48      }
49

52       * It is automatically populated with
53       * all resources defined in the project.
