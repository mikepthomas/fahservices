25  import io.swagger.jaxrs.config.BeanConfig;
26  import javax.servlet.ServletConfig;
27  import javax.servlet.ServletException;
28  import javax.servlet.annotation.WebServlet;
29  import javax.servlet.http.HttpServlet;
30  
31  /**
32   * <p>SwaggerInitConfig class.</p>
33   *
34   * @author Michael Thomas (mikepthomas@outlook.com)
35   * @version $Id: $Id
36   */
37  @WebServlet(name = "SwaggerInitConfig", loadOnStartup = 1)
38  public class SwaggerInitConfig extends HttpServlet {
39  
40      /** {@inheritDoc} */
41      @Override
42      public void init(ServletConfig servletConfig) {
43          try {
44              super.init(servletConfig);
45  
46              BeanConfig beanConfig = new BeanConfig();
47              beanConfig.setVersion("1.2");
48  
49              beanConfig.setTitle("FAHServices");
50              beanConfig.setDescription("Web services to connect to a folding@home v7 client");
51              beanConfig.setTermsOfServiceUrl("http://mikethomas.info");
52              beanConfig.setContact("mikepthomas@outlook.com");
53              beanConfig.setLicense("The GNU General Public License, Version 3");
54              beanConfig.setLicenseUrl("http://www.gnu.org/copyleft/gpl.html");
55  
56              beanConfig.setSchemes(new String[]{"http"});
57              beanConfig.setHost("localhost:8080");
58              beanConfig.setBasePath("/fahservices/rest");
59              beanConfig.setResourcePackage("info.mikethomas.fahservices.service");
60  
61              beanConfig.setScan(true);
62          } catch (ServletException e) {
63              System.out.println(e.getMessage());
64          }
65      }
66  }
